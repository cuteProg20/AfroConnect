# Use official Node.js 18 image based on Alpine or Debian
FROM node:18-alpine

# Update Alpine packages to fix vulnerabilities
RUN apk update && apk upgrade --no-cache

# Set working directory inside container
WORKDIR /app

# Copy app files
COPY . .

# Install dependencies
RUN npm install

# Expose port your app uses (e.g. 3000)
EXPOSE 3000

# Start the app
CMD ["npm", "start"]
